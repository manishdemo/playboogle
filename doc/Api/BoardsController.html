def score
  game_id = params[:id].to_i
  submitted_word = params[:word]

  # puts("game id = #{game_id} ")
  score =  get_score(submitted_word, game_id)
  render(
      json: {
          score: score
      }
  )
end

def start
  game_index = rand (0 ..  BOOGLE_STRINGS.length-1)

  render(
      json: {
          id: game_index,
          boogle_string: BOOGLE_STRINGS.at(game_index)
      }
  )
end
